1.基本用例

seajs.use(['$','arale/calendar/1.0.0/calendar','arale/calendar/1.0.0/calendar.css'],function($,calendar){
	new calendar({trigger:"#date-simple"});
})

2.设置初始值,例如初始值为2014-2-14

<input type="text" class="input length2" value="2014-2-14" id="date-initval" />

seajs.use(['$','arale/calendar/1.0.0/calendar','arale/calendar/1.0.0/calendar.css'],function($,calendar){
	new calendar({trigger:"#date-initval"});
})

<input type="text" class="input length2"  id="date-initval2" />

seajs.use(['$','arale/calendar/1.0.0/calendar','arale/calendar/1.0.0/calendar.css'],function($,calendar){
	new calendar({trigger:"#date-initval2",focus: '2014-2-15'});
})

3.限定日期选择范围

比如只能选择2014-02-02到2012-02-14之间的日期

<input type="text" class="input length2" id="date-range" value="2014-02-06" />

seajs.use(['$','arale/calendar/1.0.0/calendar','arale/calendar/1.0.0/calendar.css'],function($,calendar){
	new calendar({trigger:"#date-range",range:['2014-02-02','2014-02-14']});
})

比如只能选择2014-02-14之前的日期

<input type="text" class="input length2" id="date-range2" value="2014-02-06" />

seajs.use(['$','arale/calendar/1.0.0/calendar','arale/calendar/1.0.0/calendar.css'],function($,calendar){
	new calendar({trigger:"#date-range2",range:[null,'2014-02-14']});
})

4.触发点不是输入框

seajs.use(['$','arale/calendar/1.0.0/calendar','arale/calendar/1.0.0/calendar.css'],function($,calendar){
